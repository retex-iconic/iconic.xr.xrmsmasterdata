package com.retexspa.xr.masterdata.articolo.services.queries;

import org.springframework.stereotype.Service;

import com.retexspa.xr.masterdata.articolo.aggregates.ArticoloAggregate;

@Service
public class ArticoloQueryServiceImpl {
    
    private ArticoloAggregate articoloAggregate;

    public ArticoloQueryServiceImpl(ArticoloAggregate articoloAggregate) {
        this.articoloAggregate = articoloAggregate;
    }

    @Override
    public Iterable<ArticoloAggregate> ListOfAggregatesArticoli() {
        return articoloAggregate.findAll();
    }

}
